git init -  создает локальный репозиторий(указывает системе git , что нужно отслеживать изменения в папке)

git add название файла(или путь до файла)- добавляет файл к текущей версии
git add . - добавляет все файлы

git commit - создает версию
git commit -m 'Сообщение' -полный листинг

git log - список версий (коммитов)
git log --oneline - сокращенная версия версий

git status - отображает изменения, которые произошли после последней версии

git push название репозитория название ветки - отправка лоакльных изменений на удаленный репозиторий
git pull название репозитория название ветки- получение изменений из удаленного репозитория

git clone ссылка(SSH) - копирует репозиторий

git remote  - работа с удаленным репозиторием
git remote add < название > <url> -  добавление.привязка удаленного репозитория
Например:
git remote add origin git@github.com:usr/project.git
git remote -v  -  список удаленных репозиториев
git remote rm <название > - удаление привязки
git remote rename origin main

git reset - откат изменений или версий
git reset - откатывает все изменения
git reset <название файла>

git reset HEAD~количество - откатывает указанное количество последних коммитов(версий)

git diff - построчное отображение изменений
git diff <название файла> - просмотр изменений в конкретном файле

Работа с ветками

git branch -  просмотр имеющихся меток
git branch <название ветки> - создание ветки
git branch --delete <название ветки> или git branch --d <название ветки>  - удаление ветки

git checkout <название ветки> - переход на другую ветку
git checkout -b <название ветки> - создание и переход на эту ветку
